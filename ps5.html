# Define the "poly" class
setClass(
  "poly",
  slots = list(
    coefficients = "numeric",
    powers = "numeric"
  )
)

# Constructor
poly <- function(coefficients, powers) {
    # Stop the function when inputs are not supported
  if (!is.numeric(coefficients) || !is.numeric(powers) || length(coefficients) != length(powers) || any(powers < 0)) {
    stop("Invalid input. Coefficients and powers must be numeric vectors of the same length with only positive powers.")
  }
  new("poly", coefficients = coefficients, powers = powers)
}

# Validator
setMethod(
  "initialize",
  signature = "poly",
  definition = function(.Object, coefficients, powers) {
    .Object@coefficients <- coefficients
    .Object@powers <- powers
    .Object
  }
)

# Show method
setGeneric("show", function(object) standardGeneric("show"))
Creating a new generic function for 'show' in the global environment
[1] "show"
setMethod(
  "show",
  signature = "poly",
  function(object) {
    n <- length(object@coefficients)
    poly_str <- paste(
      object@coefficients,
      ifelse(object@powers == 0, "", ifelse(object@powers == 1, "x", paste("x^", object@powers, sep = ""))),
      sep = "",
      collapse = " + "
    )
    # Remove the "*" in the last term
    poly_str <- gsub("\\* $", "", poly_str)
    cat(poly_str, "\n")
  }
)

# Addition
setMethod(
  "+",
  signature("poly", "poly"),
  function(e1, e2) {
    max_powers <- max(max(e1@powers), max(e2@powers))
    coefficients <- numeric(max_powers + 1)
    coefficients[max_powers - e1@powers + 1] <- e1@coefficients
    coefficients[max_powers - e2@powers + 1] <- coefficients[max_powers - e2@powers + 1] + e2@coefficients
    new("poly", coefficients = coefficients, powers = max_powers:0)
  }
)

# Subtraction
setMethod(
  "-",
  signature("poly", "poly"),
  function(e1, e2) {
    max_powers <- max(max(e1@powers), max(e2@powers))
    coefficients <- numeric(max_powers + 1)
    coefficients[max_powers - e1@powers + 1] <- e1@coefficients
    coefficients[max_powers - e2@powers + 1] <- coefficients[max_powers - e2@powers + 1] - e2@coefficients
    new("poly", coefficients = coefficients, powers = max_powers:0)
  }
)
